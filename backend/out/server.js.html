/**
 * @constant {objet} express - Correspond à l'appel du module {@link https://www.npmjs.com/package/express|express}.
*/
const express = require('express')

/**
 * @constant {objet} app - Correspond à l'instanciation du module {@link https://www.npmjs.com/package/express|express}.
*/
const app = express()

const cors = require('cors')

app.use(cors())

/**
 * @constant {int} port - Correspond au port utiliser pour la création du serveur.
*/
const port = 8080

/**
 * @constant {objet} api - Correspond à l'appel du module {@link api}
*/
const api = require('./api')

/**
 * @constant {objet} bodyParser - Correspond à l'appel du module {@link https://www.npmjs.com/package/body-parser|body-parser}.
*/
const bodyParser = require("body-parser")

app.use(bodyParser.urlencoded({ extended: false }))
app.use(bodyParser.json())


/**
 * @function listen 
 * @summary Appeler pour le lancement du serveur.
 * @description La fonction nous permet de lancer le serveur soit sur le port fournie ou sur le port enrégistré dans l'environnement 
 * @param {string} port - Le port à utiliser pour la création du serveur
 * @param {function} callback - La fonction appelée à la fin du traitement, elle prends en paramètre une erreur. Si l'erreur est présent elle l'affiche sur la console, 
 *                              dans le cas contraire elle affiche la création éffectif du serveur sur la console avec sont numéro de port.
 * @throws Error listen EADDRINUSE 8080 - Implique que le port 8080 n'est plus disponible. Pour résoudre le problème vous pouvez soit déclarer une variable d'environnement nomé PORT
 *         la valeur du port souhaitée ou modifier la constant port dans le fichier server.js
 * @returns void 
*/
app.listen(process.env.PORT || port, (err) => {
    if (err) { return console.log('something bad happened', err) }
    console.log(`server is listening on ${process.env.PORT || port}`)
})

// délégation du routing de l'api au module api
app.use('/api', api)
